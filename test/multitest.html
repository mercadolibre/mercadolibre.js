MercadoLibre.init({
  xd_url: "/xd.html",
  client_id: 3096
});

MercadoLibre.post(
  "/questions",
  {
    text: $("#pregunta").val(), //texto de la pregunta
    item_id: selectedItemId //item seleccionado
  },
  function(response){
    var code = parseInt(response[0]);
    if (code <= 200 || code >= 300) {
      showErrors(response[2]);
    }
    else {
      $("<a>").modal("La pregunta se creo correctamente.Brevemente ser√° publicada").show();
    }
  }
